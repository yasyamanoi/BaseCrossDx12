BaseCrossDx12ドキュメント

【Sample001】GameObjectと単純な3D空間の描画

　このサンプルはSample001ディレクトリ内にあります。VisualStdioで該当ソリューション（VS2022でBaseCrossDx12VS2022.sln）を開いてください。
　ビルド後実行すると、以下の画面が現れます。
GameObjectについて
　ゲームのミドルウェアにおいて、画面上になんらかのオブジェクトを表示する場合、オブジェクト指向的にひとまとめにできると便利です。多くの場合GameObjectクラスといった親クラスを持ち、各オブジェクトはそこから継承して作成します。
　前のバージョンのBaseCross64においてはGameObjectクラスは多くの機能を持ち、各ゲーム制作者は、それを継承して、プレイヤーだったり敵だったりのオブジェクトを作成しました。
　しかしその構造はコンテンツ側のコードの記述量は少なく済むのですが、その反面、BaaseLibに縛られることが多く、どうしても自由度が低くなります。そのためBaseCrossDx12では、GameObjectクラスは、存在はしますが、その機能は最小限に抑えてあります。GameObjectクラスはLibs/BaseLib/Standerdにあります。
　BaseCross64でもそうでしたが、アプリケーション全体がbasecrossというnamespaceに含まれます。
　こうすることで他の環境との名前の衝突を避けることができます（完全ではないですが）。また 
	using namespace std;
	using namespace bsm;
　と記述すことで、STLとbsm計算ライブラリのnamespace名を省略できます。
　GameObjectクラスはさまざまな仮想関数があります。これらは派生クラスで多重定義可能です。
シェーダーについて
　シェーダーは、コンテンツ側のフィルタシェーダーファイルフィルタにあります。ディレクトリ的には、コンテンツ側プロジェクトのShadersに入っています。新しくシェーダーを作成する場合はビルド時にcsoファイルにコンパイルされるよう設定してください。その際csoファイルはmedia/Shadersに保存されるようにします。
　そうすることで、例えばScene.hに 
	DECLARE_DX12SHADER(BcVSPNTStaticPL)
	DECLARE_DX12SHADER(BcPSPNTPL)
	DECLARE_DX12SHADER(BcVSPNTStaticPLShadow)
	DECLARE_DX12SHADER(BcPSPNTPLShadow)
　のようにDECLARE_DX12SHADERマクロを記述し、Scene.cppに 
	IMPLEMENT_DX12SHADER(BcVSPNTStaticPL, App::GetShadersDir() + L"BcVSPNTStaticPL.cso")
	IMPLEMENT_DX12SHADER(BcPSPNTPL, App::GetShadersDir() + L"BcPSPNTPL.cso")
	IMPLEMENT_DX12SHADER(BcVSPNTStaticPLShadow, App::GetShadersDir() + L"BcVSPNTStaticPLShadow.cso")
	IMPLEMENT_DX12SHADER(BcPSPNTPLShadow, App::GetShadersDir() + L"BcPSPNTPLShadow.cso")
　とIMPLEMENT_DX12SHADERマクロを記述すれば、これらのシェーダーを描画処理に使えるようになります。
GameObjectと派生クラス
　このサンプルではGameObjectを継承して、MyObjectという、中間的な派生クラスを作っています。多くの処理をこのMyObjectで行い、土台であるSkyGroundや、くるくる回転するWallBoxクラスはMyObjectから継承しています。
MyObject::OnCreate()関数
	void MyObject::OnCreate(ID3D12GraphicsCommandList* pCommandList) {
		auto pBaseScene = BaseScene::Get();
		auto& frameResources = pBaseScene->GetFrameResources();
		auto pBaseDevice = BaseDevice::GetBaseDevice();
		//デフォルトのコンスタントバッファ
		for (size_t i = 0; i < BaseDevice::FrameCount; i++) {
			m_constantBufferIndex =
				frameResources[i]->AddBaseConstantBufferSet<BasicConstantBuffer>(pBaseDevice->GetD3D12Device());
		}
		//シャドウマップのコンスタントバッファ
		for (size_t i = 0; i < BaseDevice::FrameCount; i++) {
			m_shadowConstantBufferIndex =
				frameResources[i]->AddBaseConstantBufferSet<ShadowConstantBuffer>(pBaseDevice->GetD3D12Device());
		}
	}
　ここではコンスタントバッファ作りだしています。AddBaseConstantBufferSet()関数はBaseLibのFrameResource.hにあり、以下のような内容です。 
		template<typename T>
		size_t AddBaseConstantBufferSet(ID3D12Device* pDevice) {
			BaseConstantBufferSet cbSet;
			const CD3DX12_RANGE readRange(0, 0); // We do not intend to read from this resource on the CPU.
			auto size = m_baseConstantBufferSetVec.size();
			ThrowIfFailed(CreateConstantBuffer(pDevice, sizeof(T), &cbSet.m_baseConstantBuffer, nullptr, D3D12_RESOURCE_STATE_GENERIC_READ));
			NAME_D3D12_OBJECT(cbSet.m_baseConstantBuffer);
			ThrowIfFailed(cbSet.m_baseConstantBuffer->Map(0, &readRange, reinterpret_cast<void**>(&cbSet.m_pBaseConstantBufferWO)));
			m_baseConstantBufferSetVec.push_back(cbSet);
			return size;
		}
　これで、指定のサイズのGPU側コンスタントバッファを作り出し、そのインデックスを返します。
MyObject::UpdateConstantBuffers()関数
	void GameObject::UpdateConstantBuffers(Scene* scene) {
		//シーンのコンスタントバッファ
		{
			auto& frameResources = scene->GetFrameResources();
			auto pBaseDevice = BaseDevice::GetBaseDevice();
			auto& viewport = scene->GetViewport();

			auto myCamera = scene->GetMyCamera();
			auto myLightSet = scene->GetLightSet();

			//初期化
			m_constantBuffer = {};
			m_constantBuffer.activeFlg.x = 3;
			m_constantBuffer.activeFlg.y = 1;

			//ワールド行列の設定
			auto world = XMMatrixAffineTransformation(
				m_param.scale,
				m_param.rotOrigin,
				m_param.quaternion,
				m_param.position
			);
			auto view = (XMMATRIX)((Mat4x4)myCamera->GetViewMatrix());
			auto proj = (XMMATRIX)((Mat4x4)myCamera->GetProjMatrix());
			auto worldView = world * view;
			m_constantBuffer.worldViewProj = Mat4x4(XMMatrixTranspose(XMMatrixMultiply(worldView, proj)));
			//中略
		}
		//シャドウマップのコンスタントバッファ
		{
			auto& frameResources = scene->GetFrameResources();
			auto pBaseDevice = BaseDevice::GetBaseDevice();
			auto& viewport = scene->GetViewport();

			auto myCamera = scene->GetMyCamera();
			myCamera->CalculateMatrix();
			auto myLightSet = scene->GetLightSet();

			m_shadowConstantBuffer = {};

			auto lights = myLightSet;
			auto light = lights->GetMainBaseLight();

			//位置の取得
			Vec3 pos = m_param.position;
			Vec3 posSpan = (Vec3)light.m_directional;
			posSpan *= Vec3(Shadowmap::GetPosAdjustment());
			pos += posSpan;
			//ワールド行列の決定
			auto param_tmp = m_param;
			param_tmp.position = pos;

			auto world = XMMatrixAffineTransformation(
				param_tmp.scale,
				param_tmp.rotOrigin,
				param_tmp.quaternion,
				param_tmp.position
			);
			//中略
		}

	}
　長いので省略していますが、ここではコンスタントバッファを設定しています。
MyObject::CommitConstantBuffers()関数
　ここでは、コンスタントバッファを該当フレームのGPU側リソースに転送しています。 
	void MyObject::CommitConstantBuffers(Scene* scene) {
		auto pCurrentFrameResource = scene->GetCurrentFrameResource();
		//シーン
		memcpy(pCurrentFrameResource->m_baseConstantBufferSetVec[m_constantBufferIndex].m_pBaseConstantBufferWO,
			&m_constantBuffer, sizeof(m_constantBuffer));
		//シャドウマップ
		memcpy(pCurrentFrameResource->m_baseConstantBufferSetVec[m_shadowConstantBufferIndex].m_pBaseConstantBufferWO,
			&m_shadowConstantBuffer, sizeof(m_shadowConstantBuffer));
	}

	}
MyObject::OnSceneDraw()関数
　ここでは、シーンを描画します。 
	void MyObject::OnSceneDraw(ID3D12GraphicsCommandList* pCommandList) {

		auto pBaseScene = BaseScene::Get();
		auto& frameResources = pBaseScene->GetFrameResources();
		auto pCurrentFrameResource = pBaseScene->GetCurrentFrameResource();
		auto pBaseDevice = BaseDevice::GetBaseDevice();
		auto& viewport = pBaseScene->GetViewport();
		auto& scissorRect = pBaseScene->GetScissorRect();
		auto depthDsvs = pBaseScene->GetDepthDsvs();

		//シェーダリソースのハンドルの設定
		CD3DX12_GPU_DESCRIPTOR_HANDLE srvHandle(
			pBaseScene->GetCbvSrvUavDescriptorHeap()->GetGPUDescriptorHandleForHeapStart(),
			m_texture->GetSrvIndex(),
			pBaseScene->GetCbvSrvUavDescriptorHandleIncrementSize()
		);
		pCommandList->SetGraphicsRootDescriptorTable(pBaseScene->GetGpuSlotID(L"t1"), srvHandle);

		//Cbv
		// Set scene constant buffer.
		pCommandList->SetGraphicsRootConstantBufferView(pBaseScene->GetGpuSlotID(L"b0"),
			pCurrentFrameResource->m_baseConstantBufferSetVec[m_constantBufferIndex].m_baseConstantBuffer->GetGPUVirtualAddress()); 
		//Draw
		pCommandList->IASetVertexBuffers(0, 1, &m_mesh->GetVertexBufferView());
		pCommandList->IASetIndexBuffer(&m_mesh->GetIndexBufferView());
		pCommandList->DrawIndexedInstanced(m_mesh->GetNumIndices(), 1, 0, 0, 0);
	}
　このように、共通処理をGameObject側に書いておくことで、派生クラスの負担が減ります。

WallBoxクラス
　ここでは、OnCreate()、OnUpdate関数を記述します。 
	void WallBox::OnCreate(ID3D12GraphicsCommandList* pCommandList) {
		MyObject::OnCreate(pCommandList);
		//メッシュ
		m_mesh = BaseMesh::CreateCube(pCommandList, 1.0f);

		auto texFile = App::GetRelativeAssetsDir() + L"wall.jpg";

	}

	void WallBox::OnUpdate(double elapsedTime) {
		m_totalTime += elapsedTime;
		if (m_totalTime >= XM_2PI) {
			m_totalTime = 0.0;
		}
		Quat spanQt(Vec3(1.0f, 1.0f, 0.0f), (float)(elapsedTime * 4));
		Quat quaternion(m_param.quaternion);
		quaternion *= spanQt;
		m_param.quaternion = quaternion;
		m_param.position.x = (float)sin(m_totalTime) * 2.0f;
	}
　WallBox::OnUpdate()関数ではまずm_totalTimeにelapsedTime（前回のターンから経過時間）を足し、これがXM_2PI以上になったら0.0に初期化しています。これでm_totalTimeをもとにサイン関数（三角関数のサイン）を計算します。
　その後1ターンの差分クォータニオンをspanQtに作成し、現在のquaternionに掛けてます。これで、指定の軸に従ってちょっとだけ回転するようになります。
　また行ったり来たりの処理は、m_totalTimeをもとに 
		m_param.position.x = (float)sin(m_totalTime) * 2.0f;
　と、位置情報のx要素を変更してます。
シーンクラス（Scene.h、Scene.cpp）
　ソリューションエクスプローラを見るとBaseCrossDx12フィルタの中にソース ファイルフィルタ、およびヘッダー ファイルフィルタに、いくつかのファイルがあります。これらを記述することで、上記のような画面が表示されます。
　表示しているだけで、コントローラやキーボードに反応するわけでもありません。壁模様のオブジェクトが行ったり来たりしています、Escキーで終了させることができます。
　Scene.h、Scene.cppはシーンクラスです。シーンクラスは、表示される画面そのものと思っていただいても差し支えありません。このクラスは、ライブラリ中BaseSceneクラスを継承して作られます。
　BaseSceneクラスは、Dx12の大部分の制御を行うクラスです。かなり複雑な内容ですが、Sceneクラスでは、このサンプルで最低限必要な処理のみ多重定義して実装します。
　以下がScene.hのSceneクラスの宣言部です。 
	//--------------------------------------------------------------------------------------
	// シーン
	//--------------------------------------------------------------------------------------
	class Scene : public BaseScene
	{
		vector<shared_ptr<GameObject>> m_gameObjectvec;
	public:
		Scene(UINT frameCount, PrimDevice* pSample);
		virtual ~Scene();
		template<typename T, typename... Ts>
		shared_ptr<T> AddGameObject(ID3D12GraphicsCommandList* pCommandList, Ts&&... params) {
			try {
				auto Ptr = ObjectFactory::Create<T>(pCommandList, params...);
				m_gameObjectvec.push_back(Ptr);
				return Ptr;
			}
			catch (...) {
				throw;
			}
		}

		//アクセサ
		std::shared_ptr<PerspecCamera> GetMyCamera() const {
			return m_myCamera;
		}
		std::shared_ptr<LightSet> GetLightSet() const {
			return m_myLightSet;
		}
	protected:
		std::shared_ptr<PerspecCamera> m_myCamera;
		std::shared_ptr<LightSet> m_myLightSet;

		virtual void CreateAssetResources(ID3D12Device* pDevice, ID3D12GraphicsCommandList* pCommandList)override;
		virtual void UpdateConstantBuffers()override;
		virtual void CommitConstantBuffers()override;

		virtual void Update(double elapsedTime)override;

		virtual void ShadowPass(ID3D12GraphicsCommandList* pCommandList)override;
		virtual void ScenePass(ID3D12GraphicsCommandList* pCommandList)override;
	};
　protectedメンバーの中にいくつかの多重定義された関数があります。これらを記述することで、シーン内に必要なオブジェクトを表示することができます。
　これらの多重定義された関数は、どこかで呼び出す必要がありません。システムで必要なタイミングで呼び出されます。
CreateAssetResources()関数
　CreateAssetResources()関数は、表示すべきオブジェクトを作り出すときに呼び出されます。
　実装は以下のようになっています。 
	void Scene::CreateAssetResources(ID3D12Device* pDevice, ID3D12GraphicsCommandList* pCommandList)
	{
		m_myCamera = shared_ptr<PerspecCamera>(new PerspecCamera());
		m_myLightSet = shared_ptr<LightSet>(new LightSet());

		TransParam param;
		param.scale = Vec3(1, 1, 1);
		param.rotOrigin = Vec3(0, 0, 0);
		auto quat = XMQuaternionIdentity();
		param.quaternion = bsm::Quat(quat);
		param.position = Vec3(0.0, 5.0, 0.0f);
		AddGameObject<WallBox>(pCommandList, param);
		param.scale = Vec3(50, 1, 50);
		param.position = Vec3(0, -0.5, 0.0f);
		AddGameObject<SkyGround>(pCommandList, param);
	}
　CreateAssetResources()関数をはじめ、BaseSceneクラスから継承する関数は、こう書かなければいけない、というわけでもなく、各オブジェクトを効率よく管理できる設計であれば、サンプル通りでなくても構いません。
　このサンプルでは、あらかじめSceneクラスに 
		vector<shared_ptr<GameObject>> m_gameObjectvec;
　というGameObjectの配列を持っており、それを使って各オブジェクトを管理します。
CreateAssetResources()関数では、最初に 
		m_myCamera = shared_ptr<PerspecCamera>(new PerspecCamera());
		m_myLightSet = shared_ptr<LightSet>(new LightSet());
　という形でカメラとライトを作成しています。
　その後WallBoxクラスとSkyGroundクラスの、2つのインスタンスをAddGameObject()関数を使って作成しています。AddGameObject()関数はテンプレート関数になっていて、Scene.hに記述されている通り 
		template<typename T, typename... Ts>
		shared_ptr<T> AddGameObject(ID3D12GraphicsCommandList* pCommandList, Ts&&... params) {
			try {
				auto Ptr = ObjectFactory::Create<T>(pCommandList, params...);
				m_gameObjectvec.push_back(Ptr);
				return Ptr;
			}
			catch (...) {
				throw;
			}
		}
　という記述です。BaseCross64を使ったことがある人はわかると思いますが、GameObjectクラスはObjectInterfaceの派生クラスになっており、AddGameObject()関数を使うことで、MyObject::OnCreate()関数を自動的に呼び出すことができます。

	void Scene::Update(double elapsedTime) {
		for (auto& v : m_gameObjectvec) {
			v->OnUpdate(elapsedTime);
		}
		UpdateConstantBuffers();
		CommitConstantBuffers();

　以下はScene::ShadowPass()関数です。
　ここでは、シャドウマップが書き込めるように、OMSetRenderTargets()関数で準備しています。 
	void Scene::ShadowPass(ID3D12GraphicsCommandList* pCommandList) {
		// Set necessary state.
		//basecross
		auto rootSignature = RootSignaturePool::GetRootSignature(L"BaseCrossDefault", true);
		pCommandList->SetGraphicsRootSignature(rootSignature.Get());
		ComPtr<ID3D12PipelineState> PNTShadowmapPipelineState
			= PipelineStatePool::GetPipelineState(L"PNTShadowmap", true);
		pCommandList->IASetPrimitiveTopology(D3D_PRIMITIVE_TOPOLOGY_TRIANGLELIST);
		pCommandList->RSSetViewports(1, &m_viewport);
		pCommandList->RSSetScissorRects(1, &m_scissorRect);
		// No render target needed for the shadow pass.
		pCommandList->OMSetRenderTargets(0, nullptr, FALSE, &m_depthDsvs[DepthGenPass::Shadow]);
		//以下は各オブジェクト
		for (auto& v : m_gameObjectvec) {
			v->OnShadowDraw(pCommandList);
		}
	}
　その後、各GameObject::OnShadowDraw()関数に処理を任せます。
ScenePass()関数
　Scene::ScenePass()関数では、 
	void Scene::ScenePass(ID3D12GraphicsCommandList* pCommandList)
	{
		auto& viewport = GetViewport();
		auto& scissorRect = GetScissorRect();
		auto depthDsvs = GetDepthDsvs();

		//basecross
		auto rootSignature = RootSignaturePool::GetRootSignature(L"BaseCrossDefault", true);
		pCommandList->SetGraphicsRootSignature(rootSignature.Get());

		pCommandList->IASetPrimitiveTopology(D3D_PRIMITIVE_TOPOLOGY_TRIANGLELIST);
		pCommandList->RSSetViewports(1, &viewport);
		pCommandList->RSSetScissorRects(1, &scissorRect);
		pCommandList->SetGraphicsRootDescriptorTable(GetGpuSlotID(L"t0"), m_depthSrvGpuHandles[DepthGenPass::Shadow]);        // Set the shadow texture as an SRV.
		pCommandList->OMSetRenderTargets(1, &GetCurrentBackBufferRtvCpuHandle(), FALSE, &depthDsvs[DepthGenPass::Scene]);
		//パイプラインステートを得る
		ComPtr<ID3D12PipelineState> defaultShadowPipelineState
			= PipelineStatePool::GetPipelineState(L"BcPNTStaticShadow", true);
		pCommandList->SetPipelineState(defaultShadowPipelineState.Get());
		//Sampler
		UINT index = GetSamplerIndex(L"LinearClamp");
		if (index == UINT_MAX) {
			throw BaseException(
				L"LinearClampサンプラーが特定できません。",
				L"Scene::ScenePass()"
			);
		}
		CD3DX12_GPU_DESCRIPTOR_HANDLE samplerHandle(
			GetSamplerDescriptorHeap()->GetGPUDescriptorHandleForHeapStart(),
			index,
			GetSamplerDescriptorHandleIncrementSize()
		);
		pCommandList->SetGraphicsRootDescriptorTable(GetGpuSlotID(L"s0"), samplerHandle);

		index = GetSamplerIndex(L"ComparisonLinear");
		if (index == UINT_MAX) {
			throw BaseException(
				L"ComparisonLinearサンプラーが特定できません。",
				L"Scene::ScenePass()"
			);
		}
		CD3DX12_GPU_DESCRIPTOR_HANDLE samplerHandle2(
			GetSamplerDescriptorHeap()->GetGPUDescriptorHandleForHeapStart(),
			index,
			GetSamplerDescriptorHandleIncrementSize()
		);
		pCommandList->SetGraphicsRootDescriptorTable(GetGpuSlotID(L"s1"), samplerHandle2);

		for (auto& v : m_gameObjectvec) {
			v->OnSceneDraw(pCommandList);
		}
	}

　このサンプルでは、シンプルながらもシャドウマップやオブジェクトの動的な変化をさせる、3D描画の内容となっています。
　ここでは、オブジェクトは2つですが、これが100個になっても、大きく処理は変わりません。いろんな種類のゲームオブジェクトを作成するためにはMyObjectクラスを継承すれば作成できます。
