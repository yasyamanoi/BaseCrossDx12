BaseCrossDx12ドキュメント

【Sample001】StageとGameObjectと単純な3D空間の描画

　このサンプルはSample001ディレクトリ内にあります。VisualStdioで該当ソリューション（VS2022でBaseCrossDx12VS2022.sln）を開いてください。
　ビルド後実行すると、以下の画面が現れます。
GameObjectとStageについて
　ゲームのミドルウェアにおいて、画面上になんらかのオブジェクトを表示する場合、オブジェクト指向的にひとまとめにできると便利です。多くの場合GameObjectクラスといった親クラスを持ち、各オブジェクトはそこから継承して作成します。
　またBaseCrossDx12では、SceneとGameObjectの間に位置するオブジェクトしてStageクラスを用意してます。ゲームではタイトルステージ、ゲームステージなど様々な画面が出てきます。それを効率的に扱うには、クラスを分け、必要な時にインスタンスを構築するのが便利です。
　このサンプルでは、具体的にはStageクラスを継承したGameStageクラスとGameObjectを継承したMyObjectクラスとその子クラスを用意してます。
　では早速コードを見てみましょう。
Sceneクラス
　Sceneクラスのヘッダは以下のようになっています。 
	class Scene : public BaseScene
	{
		shared_ptr<Stage> m_activeStage;
	public:
		Scene(UINT frameCount, PrimDevice* pPrimDevice);
		virtual ~Scene();
	protected:
		virtual void CreateAssetResources(ID3D12Device* pDevice, ID3D12GraphicsCommandList* pCommandList)override;
		virtual void CreatePipelineStates(ID3D12Device* pDevice)override;
		virtual void UpdateConstantBuffers()override {}
		virtual void CommitConstantBuffers()override {}
		virtual void Update(double elapsedTime)override;
		virtual void ShadowPass(ID3D12GraphicsCommandList* pCommandList)override;
		virtual void ScenePass(ID3D12GraphicsCommandList* pCommandList)override;
		virtual void UpdateUI(std::unique_ptr<UILayer>& uiLayer)override;
	};
　shared_ptr<Stage> m_activeStage;というのはステージです。GameStageクラスが設定されます。
　ここでは前項でも紹介した仮想関数群が宣言されています。ひとつずつ見ていきます。
Scene::CreateAssetResources()関数
　この関数はオブジェクトの作成時に呼ばれる関数です。ここでは、まずGameStageを作成します。 
	void Scene::CreateAssetResources(ID3D12Device* pDevice, ID3D12GraphicsCommandList* pCommandList)
	{
		m_pTgtCommandList = pCommandList;
		m_activeStage = ObjectFactory::Create<GameStage>(pDevice);
	}
　ObjectFactory::Create()テンプレート関数はObjectInterfaceクラスを親に持つ派生クラスを作成するときに使います。内部ではインスタンスを構築して、OnCreate関数を呼び出します。ここではGameStageクラスを構築しています。
　その前のm_pTgtCommandList = pCommandList;は、現在のコマンドリストを設定しています。
　DirectX12は何かとコマンドリストを使います。構築時、シャドウマップ描画時、シーン描画時などなどです。それらをコンテンツ側で管理するのは無理があるので、m_pTgtCommandListという変数にコピーしておいて各処理ではそのコマンドリストを使います。m_pTgtCommandListはSceneクラスの親クラスであるBaseSceneクラスのメンバ変数でありpublicメンバなので、どこからでもアクセス可能です。これにScene::CreateAssetResources()関数に渡されるID3D12GraphicsCommandList* pCommandListを設定しています。
　コマンドリストが渡される関数はScene::CreateAssetResources()関数のほかにはScene::ShadowPass()、Scene::ScenePass()関数です。これらはその内部でm_pTgtCommandListを設定しています。

Scene::Update()関数
　この関数はターンごとの更新を記述します。 
	void Scene::Update(double elapsedTime) {
		if (m_activeStage) {
			m_activeStage->OnUpdate(elapsedTime);
			m_activeStage->OnUpdate2(elapsedTime);
			UpdateConstantBuffers();
			CommitConstantBuffers();
		}
	}
　ステージのOnUpdate()関数とOnUpdate2()関数を呼び出していますが、これは、それぞれのオブジェクトに2回、更新の機会を与えるためです。複数のオブジェクトを配置している場合、OnUpdate()関数で位置とかを変化させ、その結果としてほかのオブジェクトとのかかわりによっては何か動きに修正をかけたりするときにOnUpdate2()関数を使用します。
　そのあとのUpdateConstantBuffers();CommitConstantBuffers();呼び出しは自分自身のメンバ関数です。
Scene::UpdateConstantBuffers()関数

	void Scene::UpdateConstantBuffers() {

			m_activeStage->OnUpdateConstantBuffers();
		}
	}
Scene::CommitConstantBuffers()関数

Scene::ShadowPass()関数
　ここではシャドウマップの描画を行います。まず、ルートシグネチャやパイプラインステートなどの設定を行います。　そしてm_pTgtCommandList = pCommandList;で、どこからでも取得できるようにコマンドリストを設定しm_activeStage->OnShadowDraw();とステージに処理を任せます。 
	void Scene::ShadowPass(ID3D12GraphicsCommandList* pCommandList) {
		// Set necessary state.
		auto rootSignature = RootSignaturePool::GetRootSignature(L"BaseCrossDefault", true);
		pCommandList->SetGraphicsRootSignature(rootSignature.Get());
		//PipelineState
		auto shadowPipeline = PipelineStatePool::GetPipelineState(L"PNTShadowmap", true);
		pCommandList->SetPipelineState(shadowPipeline.Get());
		pCommandList->IASetPrimitiveTopology(D3D_PRIMITIVE_TOPOLOGY_TRIANGLELIST);
		pCommandList->RSSetViewports(1, &m_viewport);
		pCommandList->RSSetScissorRects(1, &m_scissorRect);
		// No render target needed for the shadow pass.
		pCommandList->OMSetRenderTargets(0, nullptr, FALSE, &m_depthDsvs[DepthGenPass::Shadow]);

		if (m_activeStage) {
			m_pTgtCommandList = pCommandList;
			m_activeStage->OnShadowDraw();
		}
	}
Scene::ScenePass()関数
　ここでもシャドウマップと同様シーン描画用の環境を整え、ステージに処理を任せます。 
	void Scene::ScenePass(ID3D12GraphicsCommandList* pCommandList)
	{
		auto& viewport = GetViewport();
		auto& scissorRect = GetScissorRect();
		auto depthDsvs = GetDepthDsvs();
		// Set necessary state.
		auto rootSignature = RootSignaturePool::GetRootSignature(L"BaseCrossDefault", true);
		pCommandList->SetGraphicsRootSignature(rootSignature.Get());
		pCommandList->IASetPrimitiveTopology(D3D_PRIMITIVE_TOPOLOGY_TRIANGLELIST);
		pCommandList->RSSetViewports(1, &viewport);
		pCommandList->RSSetScissorRects(1, &scissorRect);
		//t0: DepthGenPass::Shadow
		pCommandList->SetGraphicsRootDescriptorTable(GetGpuSlotID(L"t0"), m_depthSrvGpuHandles[DepthGenPass::Shadow]); 
		// Set the shadow texture as an SRV.
		pCommandList->OMSetRenderTargets(1, &GetCurrentBackBufferRtvCpuHandle(), FALSE, &depthDsvs[DepthGenPass::Scene]);
		//PipelineState
		auto scenePipelineState = PipelineStatePool::GetPipelineState(L"BcPNTStaticShadow", true);
		pCommandList->SetPipelineState(scenePipelineState.Get());
		//Sampler
		UINT index = GetSamplerIndex(L"LinearClamp");
		if (index == UINT_MAX) {
			throw BaseException(
				L"LinearClampサンプラーが特定できません。",
				L"Scene::ScenePass()"
			);
		}
		CD3DX12_GPU_DESCRIPTOR_HANDLE samplerHandle(
			GetSamplerDescriptorHeap()->GetGPUDescriptorHandleForHeapStart(),
			index,
			GetSamplerDescriptorHandleIncrementSize()
		);
		pCommandList->SetGraphicsRootDescriptorTable(GetGpuSlotID(L"s0"), samplerHandle);

		index = GetSamplerIndex(L"ComparisonLinear");
		if (index == UINT_MAX) {
			throw BaseException(
				L"ComparisonLinearサンプラーが特定できません。",
				L"Scene::ScenePass()"
			);
		}
		CD3DX12_GPU_DESCRIPTOR_HANDLE samplerHandle2(
			GetSamplerDescriptorHeap()->GetGPUDescriptorHandleForHeapStart(),
			index,
			GetSamplerDescriptorHandleIncrementSize()
		);
		pCommandList->SetGraphicsRootDescriptorTable(GetGpuSlotID(L"s1"), samplerHandle2);

		if (m_activeStage) {
			m_pTgtCommandList = pCommandList;
			m_activeStage->OnSceneDraw();
		}
	}
　Scene::UpdateUI()関数に関しては、前項同様です。
　さて、これでシーンの処理は終わりです。
GameStageクラス
　続いてステージの処理になります。具体的にはStageクラスを継承したGameStageクラスの実装になります。
　以下はGameStageクラスのヘッダです 
	class GameStage : public Stage {
	protected:
		shared_ptr<PerspecCamera> m_myCamera;
		shared_ptr<LightSet> m_myLightSet;
	public:
		GameStage(ID3D12Device* pDevice):
			Stage(pDevice)
		{}
		virtual ~GameStage() {}
		//アクセサ
		std::shared_ptr<PerspecCamera> GetMyCamera() const {
			return m_myCamera;
		}
		std::shared_ptr<LightSet> GetLightSet() const {
			return m_myLightSet;
		}
		virtual void OnCreate()override;
	};
　いたってシンプルなヘッダです。というのは多くの処理は親クラスのStageクラスが行っています。
　詳しくはBaseLib内のStageクラスを参照ください。
GameStage::OnCreate()関数
　この関数は、構築時に呼ばれます。内容は以下です。 
	void GameStage::OnCreate() {
		m_myCamera = shared_ptr<PerspecCamera>(new PerspecCamera());
		m_myLightSet = shared_ptr<LightSet>(new LightSet());
		TransParam param;
		param.scale = Vec3(1.0f, 1.0f, 1.0f);
		param.rotOrigin = Vec3(0.0f, 0.0f, 0.0f);
		auto quat = XMQuaternionIdentity();
		param.quaternion = bsm::Quat(quat);
		param.position = Vec3(0.0f, 5.0f, 0.0f);
		AddGameObject<WallBox>(param);
		param.scale = Vec3(50.0f, 1.0f, 50.0f);
		param.position = Vec3(0.0f, -0.5, 0.0f);
		AddGameObject<SkyGround>(param);
	}
　ここでは、土台であるSkyGroundクラスと、くるくる空中で回る物体WallBoxクラスをステージに追加します。
　AddGameObject()テンプレート関数は重要です。Stageクラスに実装されており、インスタンスを作成し、GameObjectの配列に追加するという処理を行います。
　BaseCrossDx12では、縮尺は1.0は1メートルと考えます。これは重力を加味するうえで重要です（現時点では重力は関係ないですが）。
MyObjectクラス
　ステージに配置されるオブジェクトはGameObjectクラス（BaseLibで定義）を継承して作ります。
　このサンプルでは、どちらのオブジェクトも処理はほとんど同じなので、処理をまとめられるところはMyObjectクラスで処理し、個別の処理のみ、それぞれのクラスを使います。以下ヘッダです。 
	class MyObject : public GameObject {
	protected:
		BasicConstantBuffer m_constantBuffer;
		size_t m_constantBufferIndex;

		ShadowConstantBuffer m_shadowConstantBuffer;
		size_t m_shadowConstantBufferIndex;

		//フォグが有効かどうか
		bool m_fogEnabled = true;
		//フォグの開始位置
		float m_fogStart = -25.0f;
		//フォグの終了位置
		float m_fogEnd = -40.0f;
		//フォグ色
		XMFLOAT4 m_fogColor;
		//フォグベクトル
		XMFLOAT3 m_fogVector;
		MyObject(const TransParam& param) :
			GameObject(param){
			m_fogColor = XMFLOAT4(0.8f, 0.8f, 0.8f, 1.0f);
			m_fogVector = XMFLOAT3(0.0, 0.0, 1.0f);
		}
		virtual ~MyObject() {}
		shared_ptr<BaseMesh> m_mesh;
		shared_ptr < BaseTexture> m_texture;
	public:
		virtual void OnUpdateConstantBuffers(Scene* scene, shared_ptr<Stage>& stage)override;
		virtual void OnCommitConstantBuffers(Scene* scene, shared_ptr<Stage>& stage)override;
		virtual void OnCreate()override;
		virtual void OnShadowDraw()override;
		virtual void OnSceneDraw()override;
		virtual void OnDestroy()override {}
	};
　ここでは様々なメンバ変数があります。コンスタントバッファを保持するm_constantBuffer、m_shadowConstantBuffer
　そしてそのヒープ上のインデックス、他には、メッシュやテクスチャ、フォグ情報など。このサンプルでは細かな描画が行われますが、もっとシンプルな描画を記述することもできます。
　親クラスのGameObjectはTransParamというワールド変換に使うパラメータをコン数トラクタ引数にとります。
　実装する関数は赤くなっている関数です。ひとつずつ見ていきます。
MyObject::OnUpdateConstantBuffers()関数
　この関数はステージから呼ばれます。コンスタントバッファを更新します。 
	void MyObject::OnUpdateConstantBuffers(Scene* scene, shared_ptr<Stage>& stage) {
		auto& frameResources = scene->GetFrameResources();
		auto pBaseDevice = BaseDevice::GetBaseDevice();
		auto& viewport = scene->GetViewport();
		shared_ptr<PerspecCamera> myCamera;
		shared_ptr<LightSet> myLightSet;
		auto gameStage = dynamic_pointer_cast<GameStage>(stage);
		if (gameStage) {
			myCamera = gameStage->GetMyCamera();
			myLightSet = gameStage->GetLightSet();

		}
		else {
			return;
		}
		//シーンのコンスタントバッファ
		{

			//初期化
			m_constantBuffer = {};
			m_constantBuffer.activeFlg.x = 3;
			m_constantBuffer.activeFlg.y = 1;

			//ワールド行列の設定
			auto world = XMMatrixAffineTransformation(
				m_param.scale,
				m_param.rotOrigin,
				m_param.quaternion,
				m_param.position
			);
			auto view = (XMMATRIX)((Mat4x4)myCamera->GetViewMatrix());
			auto proj = (XMMATRIX)((Mat4x4)myCamera->GetProjMatrix());
			auto worldView = world * view;
			m_constantBuffer.worldViewProj = Mat4x4(XMMatrixTranspose(XMMatrixMultiply(worldView, proj)));

			//中略


		}
		//シャドウマップのコンスタントバッファ
		{

			m_shadowConstantBuffer = {};

			auto lights = myLightSet;
			auto light = lights->GetMainBaseLight();

			//位置の取得
			Vec3 pos = m_param.position;
			Vec3 posSpan = (Vec3)light.m_directional;
			posSpan *= Vec3(Shadowmap::GetPosAdjustment());
			pos += posSpan;
			//ワールド行列の決定
			auto param_tmp = m_param;
			param_tmp.position = pos;

			auto world = XMMatrixAffineTransformation(
				param_tmp.scale,
				param_tmp.rotOrigin,
				param_tmp.quaternion,
				param_tmp.position
			);

			//中略

		}
	}
　ソースコードを見るとわかりますが、かなり長いソースになります。これはこのサンプルのコンスタントバッファが複雑なためで、もっとシンプルにすることも可能です。
　この関数では、シーンのコンスタントバッファであるm_constantBufferと、シャドウマップのコンスタントバッファであるm_shadowConstantBufferを更新します。
MyObject::OnCommitConstantBuffers()関数
　設定したコンスタントバッファはシェーダーに渡さなければなりません。MyObject::OnCommitConstantBuffers()関数はその役割を担います。 
	void MyObject::OnCommitConstantBuffers(Scene* scene, shared_ptr<Stage>& stage) {
		auto pCurrentFrameResource = scene->GetCurrentFrameResource();
		//シーン
		memcpy(pCurrentFrameResource->m_baseConstantBufferSetVec[m_constantBufferIndex].m_pBaseConstantBufferWO,
			&m_constantBuffer, sizeof(m_constantBuffer));
		//シャドウマップ
		memcpy(pCurrentFrameResource->m_baseConstantBufferSetVec[m_shadowConstantBufferIndex].m_pBaseConstantBufferWO,
			&m_shadowConstantBuffer, sizeof(m_shadowConstantBuffer));
	}
　ここでは、現在のフレームのインデックス、すなわちm_constantBufferIndexとm_shadowConstantBufferIndexに先ほど更新したコンスタントバッファをコピーします。ここで使ってるm_constantBufferIndexとm_shadowConstantBufferIndexは、続いて紹介するMyObject::OnCreate()関数で設定します。
MyObject::OnCreate()関数
　ここではMyObject::OnCommitConstantBuffers()関数で使用するコンスタントバッファのフレームごとのインデックスを初期化します。 
	void MyObject::OnCreate() {
		ID3D12GraphicsCommandList* pCommandList = BaseScene::Get()->m_pTgtCommandList;
		auto pBaseScene = BaseScene::Get();
		auto& frameResources = pBaseScene->GetFrameResources();
		auto pBaseDevice = BaseDevice::GetBaseDevice();
		//デフォルトのコンスタントバッファ
		for (size_t i = 0; i < BaseDevice::FrameCount; i++) {
			m_constantBufferIndex =
				frameResources[i]->AddBaseConstantBufferSet<BasicConstantBuffer>(pBaseDevice->GetD3D12Device());
		}
		//シャドウマップのコンスタントバッファ
		for (size_t i = 0; i < BaseDevice::FrameCount; i++) {
			m_shadowConstantBufferIndex =
				frameResources[i]->AddBaseConstantBufferSet<ShadowConstantBuffer>(pBaseDevice->GetD3D12Device());
		}
	}
　ここでは 
		for (size_t i = 0; i < BaseDevice::FrameCount; i++) {
			m_constantBufferIndex =
				frameResources[i]->AddBaseConstantBufferSet<BasicConstantBuffer>(pBaseDevice->GetD3D12Device());
		}
　などで、各フレームごとのコンスタントバッファを作り出し、それをm_constantBufferIndexに上書きしてますが、各フレームのコンスタントバッファの配置は同じなので問題ありません。
MyObject::OnShadowDraw()関数
　ここではシャドウマップの描画を行います。 
	void MyObject::OnShadowDraw() {
		ID3D12GraphicsCommandList* pCommandList = BaseScene::Get()->m_pTgtCommandList;
		auto pBaseScene = BaseScene::Get();
		auto pCurrentFrameResource = pBaseScene->GetCurrentFrameResource();
		//Cbv
		// Set shadow constant buffer.
		pCommandList->SetGraphicsRootConstantBufferView(pBaseScene->GetGpuSlotID(L"b0"),
			pCurrentFrameResource->m_baseConstantBufferSetVec[m_shadowConstantBufferIndex].m_baseConstantBuffer->GetGPUVirtualAddress());
		// Draw
		pCommandList->IASetVertexBuffers(0, 1, &m_mesh->GetVertexBufferView());
		pCommandList->IASetIndexBuffer(&m_mesh->GetIndexBufferView());
		pCommandList->DrawIndexedInstanced(m_mesh->GetNumIndices(), 1, 0, 0, 0);
	}
　基本的なパイプライン設定はSceneクラスで設定済みなので、ここでは個別のオブジェクトの描画になります。
　コンスタントバッファを設定して、描画、という流れです。
MyObject::OnSceneDraw()関数
　ここではシーンの描画です。テクスチャとコンスタントバッファの設定して、描画、です。 
	void MyObject::OnSceneDraw() {
		ID3D12GraphicsCommandList* pCommandList = BaseScene::Get()->m_pTgtCommandList;
		auto pBaseScene = BaseScene::Get();
		auto& frameResources = pBaseScene->GetFrameResources();
		auto pCurrentFrameResource = pBaseScene->GetCurrentFrameResource();
		auto pBaseDevice = BaseDevice::GetBaseDevice();
		auto& viewport = pBaseScene->GetViewport();
		auto& scissorRect = pBaseScene->GetScissorRect();
		auto depthDsvs = pBaseScene->GetDepthDsvs();
		//シェーダリソース（テクスチャ）のハンドルの設定
		CD3DX12_GPU_DESCRIPTOR_HANDLE srvHandle(
			pBaseScene->GetCbvSrvUavDescriptorHeap()->GetGPUDescriptorHandleForHeapStart(),
			m_texture->GetSrvIndex(),
			pBaseScene->GetCbvSrvUavDescriptorHandleIncrementSize()
		);
		pCommandList->SetGraphicsRootDescriptorTable(pBaseScene->GetGpuSlotID(L"t1"), srvHandle);
		//Cbv
		// Set scene constant buffer.
		pCommandList->SetGraphicsRootConstantBufferView(pBaseScene->GetGpuSlotID(L"b0"),
			pCurrentFrameResource->m_baseConstantBufferSetVec[m_constantBufferIndex].m_baseConstantBuffer->GetGPUVirtualAddress());
		//Draw
		pCommandList->IASetVertexBuffers(0, 1, &m_mesh->GetVertexBufferView());
		pCommandList->IASetIndexBuffer(&m_mesh->GetIndexBufferView());
		pCommandList->DrawIndexedInstanced(m_mesh->GetNumIndices(), 1, 0, 0, 0);
	}

WallBoxクラス
　最後にWallBoxクラスです。このオブジェクトはくるくる回る処理が必要です。まず、ヘッダです。 
	class WallBox : public  MyObject {
		double m_totalTime;
	protected:
	public:
		WallBox(const TransParam& param);
		virtual ~WallBox();
		virtual void OnCreate();
		virtual void OnUpdate(double elapsedTime);
	};

WallBox::OnUpdate()関数
　このオブジェクトはさらにOnUpdate()関数の処理を記述します。 
	void WallBox::OnUpdate(double elapsedTime) {
		m_totalTime += elapsedTime;
		if (m_totalTime >= XM_2PI) {
			m_totalTime = 0.0;
		}
		Quat spanQt(Vec3(1.0f, 1.0f, 0.0f), (float)(elapsedTime * 4));
		Quat quaternion(m_param.quaternion);
		quaternion *= spanQt;
		m_param.quaternion = quaternion;
		m_param.position.x = (float)sin(m_totalTime) * 2.0f;
	}
　ここではspanQtというクォータニオン変数を作り、それを今のクォータニオンに掛け算することで回転を実装します。
　また行ったり来たりは、三角関数のsin（サイン）の計算を使用します。
